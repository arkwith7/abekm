## 2. 프론트엔드 메뉴 구조

### 2.1 현재 구조 vs 개편안

#### 현재 구조 (`/user`)

```
/user
├── 대시보드 (Dashboard)
├── 내 지식 (MyKnowledge)
├── 탐색 (ContainerExplorer)
├── 검색 (SearchPage)
├── IP 포트폴리오 (IPPortfolioPage)
├── IP Copilot (AgentChatPage)
├── Agent Chat (AgentChatPage)
├── PPT Agent (PresentationAgentChatPage)
├── 대화 이력 (ChatHistoryPage)
├── 프로필 (UserProfilePage)
├── 권한 요청 (PermissionRequestsPage)
└── 설정 (UserSettingsPage)
```

#### 개편안 (`/ip`)

```
/ip
├── 🏠 IP 인사이트 대시보드
│   ├── 보유 특허 현황 카드
│   ├── 금주 조사 현황 카드
│   ├── 포트폴리오 가치 카드
│   ├── 리스크 알림 카드
│   └── 빠른 시작 위젯
│
├── 📁 IP 포트폴리오
```

```typescript
// components/ip/MetricCard.tsx

interface MetricCardProps {
  title: string;
  value: string | number;
  subValue?: string;
  trend?: {
    direction: 'up' | 'down' | 'neutral';
    value: string;
  };
  icon?: React.ReactNode;
  action?: {
    label: string;
    onClick: () => void;
  };
}

const MetricCard: React.FC<MetricCardProps> = ({
  title,
  value,
  subValue,
  trend,
  icon,
  action,
}) => {
  return (
    <div className="bg-white rounded-lg shadow-sm border p-6">
      <div className="flex items-center justify-between">
        <div>
          <p className="text-sm text-gray-500">{title}</p>
          <p className="text-2xl font-bold mt-1">{value}</p>
          {subValue && (
            <p className="text-sm text-gray-400 mt-1">{subValue}</p>
          )}
          {trend && (
            <div className={`flex items-center mt-2 ${
              trend.direction === 'up' ? 'text-green-500' :
              trend.direction === 'down' ? 'text-red-500' : 'text-gray-500'
            }`}>
              {trend.direction === 'up' && <ArrowUpIcon className="w-4 h-4" />}
              {trend.direction === 'down' && <ArrowDownIcon className="w-4 h-4" />}
              <span className="text-sm ml-1">{trend.value}</span>
            </div>
          )}
        </div>
        {icon && (
          <div className="text-indigo-500">{icon}</div>
        )}
      </div>
      {action && (
        <button
          onClick={action.onClick}
          className="mt-4 text-sm text-indigo-600 hover:text-indigo-800"
        >
          {action.label} →
        </button>
      )}
    </div>
  );
};
```

### 5.3 API 연동

```typescript
// services/ipService.ts

export const ipService = {
  // 대시보드 메트릭
  getDashboardMetrics: () => 
    api.get('/api/v1/ip/dashboard/metrics'),
  
  // 선행기술 조사
  startPriorArtSearch: (data: PriorArtRequest) =>
    api.post('/api/v1/agent/chat/stream', {
      ...data,
      agent_type: 'prior_art',
    }),
  
  // 조사 결과
  getSearchResults: (sessionId: string) =>
    api.get(`/api/v1/agent/sessions/${sessionId}/results`),
  
  // X-Y 매트릭스
  getXYMatrix: (sessionId: string) =>
    api.get(`/api/v1/agent/sessions/${sessionId}/xy-matrix`),
  
  // 보고서 다운로드
  downloadReport: (sessionId: string, format: 'pdf' | 'docx') =>
    api.get(`/api/v1/agent/sessions/${sessionId}/report`, {
      params: { format },
      responseType: 'blob',
    }),
};
```
